💻 Code Source (Firmware Arduino)

Ce script gère l'initialisation de l'affichage parallèle et cadence les impulsions du sonar.

#include <LiquidCrystal.h>

// Configuration des broches LCD (RS, E, D4, D5, D6, D7)
LiquidCrystal lcd(12, 11, 5, 4, 3, 2);
const int pirPin = 7;
const int trigPin = 9;
const int echoPin = 8;

void setup() {
  lcd.begin(16, 2);
  pinMode(pirPin, INPUT);
  pinMode(trigPin, OUTPUT);
  pinMode(echoPin, INPUT);
}
